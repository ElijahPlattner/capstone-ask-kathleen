# Ask Kathleen - Local Flask + Frontend

## Setup (macOS / Linux)
python -m venv venv
source venv/bin/activate
pip install -r requirements.txt
python app_flask.py
# open http://localhost:8000

## Setup (Windows PowerShell)
python -m venv venv
venv\Scripts\Activate.ps1
pip install -r requirements.txt
python app_flask.py
# open http://localhost:8000
